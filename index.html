    <script>

        // 1. CONFIGURATION

        // =========================================================================================


        // IMPORTANT: Set the time when the survey should EXPIRE.

        // Use the format: 'YYYY-MM-DDTHH:MM:SS-OFFSET'

        // Example: If your meeting is TODAY, Nov 18, 2025, and closes at 10:30 PM EST (UTC-05)

        // Set it to: '2025-11-18T22:30:00-05:00'

        // If you're unsure of the time zone, use UTC time (Z at the end).


        // --- USER CONFIGURABLE VARIABLES ---

        const EXPIRATION_TIME_UTC = '2025-11-18T22:30:00-05:00'; // Set to 10:30 PM EST

        const SURVEY_URL = 'https://docs.google.com/forms/d/e/1FAIpQLSfB_Survey_Link/viewform'; // REPLACE with your actual survey URL

        // --- END USER CONFIGURABLE VARIABLES ---


        // 2. LOGIC

        // =========================================================================================


        document.addEventListener('DOMContentLoaded', () => {

            const loadingState = document.getElementById('loading-state');

            const closedState = document.getElementById('closed-state');

            const expirationDate = new Date(EXPIRATION_TIME_UTC);

            const now = new Date();


            const isExpired = now.getTime() >= expirationDate.getTime();


            if (isExpired) {

                // If expired, show the closed state

                loadingState.classList.add('hidden');

                closedState.classList.remove('hidden');

                document.body.classList.remove('bg-gray-50');

                document.body.classList.add('bg-red-50');

                console.log('Attendance window is CLOSED. Expiration time:', expirationDate.toISOString());

            } else {

                // If still open, log and redirect immediately

                console.log('Attendance window is OPEN. Redirecting to survey...');

                console.log('Time left until:', expirationDate.toISOString());

                

                // Add a small delay for the user to see "Checking..." before redirecting

                setTimeout(() => {

                    window.location.replace(SURVEY_URL);

                }, 500); 

            }

        });

    </script>
